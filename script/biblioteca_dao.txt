
CREATE TABLE Editore(
CodiceE VARCHAR(10) PRIMARY KEY,
Nome VARCHAR(30)
);


CREATE TABLE Autore (
  CodiceA INT PRIMARY KEY,
  NomeA VARCHAR(20),
  CognomeA VARCHAR(20),
  AnnoN INT,
  AnnoM INT,
  Sesso VARCHAR(1),
  Nazione VARCHAR(20)
);


CREATE TABLE Genere(
CodiceG VARCHAR(10) PRIMARY KEY,
Descrizione VARCHAR(30)
);


CREATE TABLE libro(
  CodiceR INT PRIMARY KEY,
  Titolo VARCHAR(255),
  NumPag INT,
  Anno INT,
  CodiceA INT,
  CodiceE VARCHAR(10),
  CodiceG INT,
  FOREIGN KEY (CodiceA) REFERENCES autore(CodiceA),
  FOREIGN KEY (CodiceE) REFERENCES editore(CodiceE),
  FOREIGN KEY (CodiceG) REFERENCES genere(CodiceG)
);
